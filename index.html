function findIndex(array, value) {  
  let first = 0;  
  let last = array.length - 1;  
  let between = Math.round(last / 2);  
  while (array[between] !== value 
  && first < last) {  
    if (array[between] < value) {  
      first = between + 1;  
    } 
    else if (array[between] > value) {  
      last = between - 1;  
    }  
    between = Math.round(
        (first + last) / 2
    );  
  }  
  return Math.abs(between);  
}
